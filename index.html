const express = require('express');
const mongoose = require('mongoose');
const cors = require('cors');
require('dotenv').config();

const app = express();
const PORT = process.env.PORT || 5000;

// Middleware
app.use(cors());
app.use(express.json());

// MongoDB Connection
mongoose.connect(process.env.MONGO_URI, {
  useNewUrlParser: true,
  useUnifiedTopology: true
})
.then(() => console.log('MongoDB connected'))
.catch(err => console.error(err));

// Product Schema
const productSchema = new mongoose.Schema({
  name: String,
  price: Number,
  description: String,
  image: String,
  category: String
});

const Product = mongoose.model('Product', productSchema);

// Routes
app.get('/api/products', async (req, res) => {
  try {
    const products = await Product.find();
    res.json(products);
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

app.post('/api/products', async (req, res) => {
  try {
    const newProduct = new Product(req.body);
    await newProduct.save();
    res.status(201).json(newProduct);
  } catch (error) {
    res.status(400).json({ error: error.message });
  }
});

app.listen(PORT, () => {
  console.log(`Server running on port ${PORT}`);
});

// Weather Dashboard Application
class WeatherDashboard {
  constructor() {
    this.apiKey = 'YOUR_API_KEY';
    this.baseUrl = 'https://api.openweathermap.org/data/2.5';
    this.cache = new Map();
    
    this.searchInput = document.getElementById('search');
    this.cityName = document.getElementById('city-name');
    this.tempDisplay = document.getElementById('temperature');
    
    this.init();
  }
  
  async init() {
    if (navigator.geolocation) {
      navigator.geolocation.getCurrentPosition(
        position => this.getWeatherByCoords(position.coords),
        () => this.setDefaultCity()
      );
    }
    
    this.searchInput.addEventListener('keypress', (e) => {
      if (e.key === 'Enter') this.searchCity();
    });
  }
  
  async searchCity() {
    const city = this.searchInput.value.trim();
    if (!city) return;
    
    try {
      const data = await this.fetchWeatherData(city);
      this.updateDisplay(data);
    } catch (error) {
      this.showError('City not found');
    }
  }
  
  async fetchWeatherData(city) {
    const response = await fetch(
      `${this.baseUrl}/weather?q=${city}&units=metric&appid=${this.apiKey}`
    );
    
    if (!response.ok) {
      throw new Error('City not found');
    }
    
    return response.json();
  }
  
  updateDisplay(data) {
    const { name, main, sys } = data;
    this.cityName.textContent = `${name}, ${sys.country}`;
    this.tempDisplay.textContent = `${Math.round(main.temp)}Â°C`;
  }
  
  showError(message) {
    alert(message);
  }
}

document.addEventListener('DOMContentLoaded', () => {
  new WeatherDashboard();
});
// Weather API Service Module

const API_CONFIG = {
  baseUrl: 'https://api.openweathermap.org/data/2.5',
  oneCallUrl: 'https://api.openweathermap.org/data/3.0/onecall',
  iconsUrl: 'https://openweathermap.org/img/wn'
};

class WeatherAPI {
  constructor(apiKey) {
    this.apiKey = apiKey;
  }
  
  async getCurrentWeather(city) {
    const response = await fetch(
      `${API_CONFIG.baseUrl}/weather?q=${city}&units=metric&appid=${this.apiKey}`
    );
    
    if (!response.ok) {
      throw new Error(`HTTP error! status: ${response.status}`);
    }
    
    return response.json();
  }
  
  async getForecast(city) {
    const response = await fetch(
      `${API_CONFIG.baseUrl}/forecast?q=${city}&units=metric&appid=${this.apiKey}`
    );
    
    if (!response.ok) {
      throw new Error(`HTTP error! status: ${response.status}`);
    }
    
    return response.json();
  }
  
  getIconUrl(iconCode) {
    return `${API_CONFIG.iconsUrl}/${iconCode}@2x.png`;
  }
}

export default WeatherAPI;
